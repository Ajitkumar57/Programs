


//Reverse of String
function reverseString(str: String) {
    let reversed = "";
    for (let i = str.length - 1; i >= 0; i--) {
        reversed += str[i];
    }
    return reversed;
}
console.log(reverseString("ajitkumar"));

//Reverse Number
function reverseNumber(num: number) {
    let rev = 0;
    while (num > 0) {
        let digit = num % 10;
        rev = rev * 10 + digit;
        num = (num - digit) / 10;
    }
    return rev;
}
console.log(reverseNumber(12345))

//last non repating character in String
function lastNonRepeatingChar(str: string): string | null {
  for (let i = str.length - 1; i >= 0; i--) {
    let count = 0;

    // check how many times str[i] appears
    for (let j = 0; j < str.length; j++) {
      if (str[i] === str[j]) {
        count++;
      }
    }

    // if only once, return it
    if (count === 1) {
      return str[i];
    }
  }
  return null;
}
